# еҶ…зҪ®еҜ№иұЎ

# еұһжҖ§жҸҸиҝ°з¬Ұ

  • д»ҺES5ејҖе§ӢпјҢжүҖжңүзҡ„еұһжҖ§йғҪе…·еӨҮдәҶеұһжҖ§жҸҸиҝ°з¬ҰгҖӮ
    var myObject = {
  a: 2
};

const res = Object.getOwnPropertyDescriptor(myObject, 'a');
console.log(res);
// {
//   value: 2,
//   writable: true,
//   enumerable: true,
//   configurable: true
// }
  • ж–°е»әдёҖдёӘеұһжҖ§жҲ–иҖ…дҝ®ж”№дёҖдёӘе·ІжңүеұһжҖ§пјҲеҰӮжһңе®ғжҳҜconfigurableпјүгҖӮ
    var myObject = {};

Object.defineProperty(myObject, 'a', {
  value:2,
  writable: true,
  enumerable: true,
  configurable: true
});

console.log(myObject);
// { a: 2 }

# дёҚеҸҳжҖ§

  • еҜ№иұЎеёёйҮҸпјӣ з»“еҗҲwritable:falseе’Ңconfigurable:falseе°ұеҸҜд»ҘеҲӣе»әдёҖдёӘзңҹжӯЈзҡ„еёёйҮҸеұһжҖ§пјҲдёҚеҸҜдҝ®ж”№гҖҒйҮҚе®ҡд№үжҲ–иҖ…еҲ йҷӨпјүгҖӮ
  • зҰҒжӯўжү©еұ•пјӣ зҰҒжӯўдёҖдёӘеҜ№иұЎж·»еҠ ж–°еұһжҖ§е№¶дё”дҝқз•ҷе·ІжңүеұһжҖ§пјҢеҸҜд»ҘдҪҝз”ЁObject.preventExtensions(..)гҖӮ
  • еҜҶе°Ғпјӣ
    Object.seal(..)дјҡеҲӣе»әдёҖдёӘвҖңеҜҶе°ҒвҖқзҡ„еҜ№иұЎпјҢиҝҷдёӘж–№жі•е®һйҷ…дёҠдјҡеңЁдёҖдёӘзҺ°жңүеҜ№иұЎдёҠи°ғз”ЁObject.preventExtensions(..)е№¶жҠҠжүҖжңүзҺ°жңүеұһжҖ§ж Үи®°дёәconfigurable:falseгҖӮдёҚиғҪж·»еҠ ж–°еұһжҖ§пјӣдёҚиғҪйҮҚж–°й…ҚзҪ®жҲ–иҖ…еҲ йҷӨд»»дҪ•зҺ°жңүеұһжҖ§пјӣеҸҜд»Ҙдҝ®ж”№еұһжҖ§зҡ„еҖјгҖӮ
  • еҶ»з»“пјӣ
    Object.freeze(..)дјҡеҲӣе»әдёҖдёӘеҶ»з»“еҜ№иұЎпјҢиҝҷдёӘж–№жі•е®һйҷ…дёҠдјҡеңЁдёҖдёӘзҺ°жңүеҜ№иұЎдёҠи°ғз”ЁObject.seal(..)е№¶жҠҠжүҖжңүвҖңж•°жҚ®и®ҝй—®вҖқеұһжҖ§ж Үи®°дёәwritable:falseпјҢиҝҷж ·е°ұж— жі•дҝ®ж”№е®ғд»¬зҡ„еҖјгҖӮиҝҷдёӘеҜ№иұЎеј•з”Ёзҡ„е…¶д»–еҜ№иұЎжҳҜдёҚеҸ—еҪұе“Қзҡ„гҖӮ

    вҖңж·ұеәҰеҶ»з»“вҖқдёҖдёӘеҜ№иұЎ

    йҒҚеҺҶе®ғеј•з”Ёзҡ„жүҖжңүеҜ№иұЎе№¶еңЁиҝҷдәӣеҜ№иұЎдёҠи°ғз”ЁObject.freeze(..)гҖӮдҪҶеҸҜиғҪдјҡеңЁж— ж„ҸдёӯеҶ»з»“е…¶д»–пјҲе…ұдә«пјүеҜ№иұЎгҖӮ

# [[Get]]

# дҫӢпјҡmyObject.a

е®һйҷ…дёҠжҳҜеңЁmyObjectдёҠе®һзҺ°дәҶ[[Get]]ж“ҚдҪңпјҲжңүзӮ№еғҸеҮҪж•°и°ғз”Ёпјҡ[[Get]]()пјүгҖӮ

  1. йҰ–е…ҲеңЁеҜ№иұЎдёӯжҹҘжүҫжҳҜеҗҰжңүеҗҚз§°зӣёеҗҢзҡ„еұһжҖ§пјӣ
  2. жүҫеҲ°е°ұдјҡиҝ”еӣһиҝҷдёӘеұһжҖ§зҡ„еҖјпјӣ
  3. жІЎжңүжүҫеҲ°еҗҚз§°зӣёеҗҢзҡ„еұһжҖ§пјҢйҒҚеҺҶеҸҜиғҪеӯҳеңЁзҡ„[[Prototype]]й“ҫпјҢд№ҹе°ұжҳҜеҺҹеһӢй“ҫпјӣ
  4. еұһжҖ§дёҚеӯҳеңЁпјҢйӮЈ[[Get]]ж“ҚдҪңдјҡиҝ”еӣһеҖјundefinedгҖӮ

жңҖйҮҚиҰҒзҡ„еӣ зҙ пјҡеҜ№иұЎдёӯжҳҜеҗҰе·Із»ҸеӯҳеңЁиҝҷдёӘеұһжҖ§

  • е·Із»ҸеӯҳеңЁпјӣ
    1. еұһжҖ§жҳҜеҗҰжҳҜи®ҝй—®жҸҸиҝ°з¬ҰпјҹжҳҜе№¶дё”еӯҳеңЁsetterе°ұи°ғз”Ёsetterпјӣ
    2. еұһжҖ§зҡ„ж•°жҚ®жҸҸиҝ°з¬ҰдёӯwritableжҳҜеҗҰжҳҜfalseпјҹеҰӮжһңжҳҜпјҢеңЁйқһдёҘж јжЁЎејҸдёӢйқҷй»ҳеӨұиҙҘпјҢеңЁдёҘж јжЁЎејҸдёӢжҠӣеҮәTypeErrorејӮеёёпјӣ
    3. еҰӮжһңйғҪдёҚжҳҜпјҢе°ҶиҜҘеҖји®ҫзҪ®дёәеұһжҖ§зҡ„еҖјгҖӮ
  • дёҚеӯҳеңЁгҖӮ
    [[Put]]ж“ҚдҪңдјҡжӣҙеҠ еӨҚжқӮпјҢдёҺ[[Prototype]]жңүе…ігҖӮ

  • Object.getOwnPropertyNames(..)дјҡиҝ”еӣһдёҖдёӘж•°з»„пјҢеҢ…еҗ«жүҖжңүеұһжҖ§пјҢж— и®әе®ғд»¬жҳҜеҗҰеҸҜжһҡдёҫпјҲдёҚеҢ…жӢ¬еҺҹеһӢй“ҫдёҠеұһжҖ§пјүпјӣ

# йҒҚеҺҶ

жҷ®йҖҡзҡ„еҜ№иұЎжІЎжңүеҶ…зҪ®зҡ„@@iteratorпјҢж— жі•иҮӘеҠЁе®ҢжҲҗfor..ofйҒҚеҺҶпјҲдёәдәҶйҒҝе…ҚеҪұе“ҚжңӘжқҘзҡ„еҜ№иұЎзұ»еһӢпјүгҖӮ
з»ҷеҜ№иұЎе®ҡд№ү@@iteratorпјҡ

var myObject = {
  a: 2,
  b: 3,
  c: 4
}

// Symbol.iterator дёәжҜҸдёҖдёӘеҜ№иұЎе®ҡд№үдәҶй»ҳи®Өзҡ„иҝӯд»ЈеҷЁгҖӮиҜҘиҝӯд»ЈеҷЁеҸҜд»Ҙиў« for...of еҫӘзҺҜдҪҝз”ЁгҖӮ
Object.defineProperty(myObject, Symbol.iterator, {
  enumerable: true,
  writable: true,
  configurable: true,
  value: function() {
    var o = this
    var idx = 0
    var ks = Object.keys(o)
    return {
      next: function() {
        return {
          value: { key: ks[idx], value: o[ks[idx++]] }, // idx++жҳҜе…ҲиөӢеҖјз»ҷidxеҗҺ++
          done: idx > ks.length
        };
      }
    };
  }
});

// жүӢеҠЁйҒҚеҺҶ
var obj = myObject[Symbol.iterator]();
const res = obj.next();
const res1 = obj.next();
const res2 = obj.next();
const res3 = obj.next();
console.log(res);
console.log(res1);
console.log(res2);
console.log(res3);

// for...ofйҒҚеҺҶ
for (var value of myObject) {
  console.log(value);
}


// { value: { key: 'a', value: 2 }, done: false }
// { value: { key: 'b', value: 3 }, done: false }
// { value: { key: 'c', value: 4 }, done: false }
// { value: { key: undefined, value: undefined }, done: true }
// { key: 'a', value: 2 }
// { key: 'b', value: 3 }
// { key: 'c', value: 4 }
